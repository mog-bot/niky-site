(() => { // --------- Globals & Utils ---------- const canvas = document.getElementById('game'); const ctx = canvas.getContext('2d', { alpha: false }); let W = canvas.width = innerWidth; let H = canvas.height = innerHeight; window.addEventListener('resize', () => { W = canvas.width = innerWidth; H = canvas.height = innerHeight; }); function clamp(v, a, b){ return Math.max(a, Math.min(b, v)); } function lerp(a,b,t){ return a + (b-a)*t; } // --------- Input ---------- const Input = { up:false, down:false, left:false, right:false }; window.addEventListener('keydown', e=>{ if(e.key === 'w' || e.key === 'ArrowUp') Input.up = true; if(e.key === 's' || e.key === 'ArrowDown') Input.down = true; if(e.key === 'a' || e.key === 'ArrowLeft') Input.left = true; if(e.key === 'd' || e.key === 'ArrowRight') Input.right = true; if(e.key === 't') Game.toggleWaves(); }); window.addEventListener('keyup', e=>{ if(e.key === 'w' || e.key === 'ArrowUp') Input.up = false; if(e.key === 's' || e.key === 'ArrowDown') Input.down = false; if(e.key === 'a' || e.key === 'ArrowLeft') Input.left = false; if(e.key === 'd' || e.key === 'ArrowRight') Input.right = false; }); // --------- Camera ---------- const Camera = { x:0, y:0, follow(targetX, targetY){ // ease-follow this.x = lerp(this.x, targetX - W*0.35, 0.08); this.y = lerp(this.y, targetY - H*0.5, 0.08); } }; // --------- Terrain (simple procedural hills above water) ---------- const Terrain = { step: 40, amp: 120, seed: 0.5, cache: {}, heightAt(x){ // simple layered sin / noise-ish function => stable, quick // x in world coords const k = 0.002; return Math.sin((x*k) + (this.seed * 6.28)) * this.amp + Math.sin((x*k*0.3) + this.seed*12) * (this.amp*0.25) + 120; // base ground offset }, draw(ctx, camX, camY){ ctx.save(); ctx.translate(-camX, -camY); // draw waterline and ground above // ground shape ctx.beginPath(); const left = camX - 200; const right = camX + W + 200; ctx.moveTo(left, H+100); for(let x = left; x <= right; x += this.step){ ctx.lineTo(x, H - this.heightAt(x)); } ctx.lineTo(right, H+100); ctx.closePath(); ctx.fillStyle = '#5b8a37'; ctx.fill(); // simple obstacles: draw some ramps for(let x = left + 300; x < right; x += 800){ const gx = x; const gy = H - this.heightAt(gx); ctx.fillStyle = '#7f5a3b'; ctx.fillRect(gx, gy - 16, 80, 16); } ctx.restore(); } }; // --------- Waves ---------- const Wave = { enabled: false, amplitude: 18, wavelength: 220, speed: 0.0015, // phase speed t: 0, toggle(){ this.enabled = !this.enabled; }, update(dt){ this.t += dt; }, heightAt(x){ if(!this.enabled) return H - 60; // flat water y return (H - 60) + Math.sin( (x/this.wavelength) + (this.t * 6.28 * this.speed) ) * this.amplitude; }, draw(ctx, camX, camY){ ctx.save(); ctx.translate(-camX, -camY); // draw water surface across screen ctx.beginPath(); const left = camX - 200; const right = camX + W + 200; ctx.moveTo(left, H+200); for(let x = left; x <= right; x += 10){ ctx.lineTo(x, this.heightAt(x)); } ctx.lineTo(right, H+200); ctx.closePath(); const grad = ctx.createLinearGradient(0,0,0,H); grad.addColorStop(0, '#4dbef3'); grad.addColorStop(1, '#1b88c3'); ctx.fillStyle = grad; ctx.fill(); ctx.restore(); } }; // --------- Boat ---------- class Boat { constructor(){ this.x = 120; this.y = Wave.heightAt ? Wave.heightAt(120) - 20 : H - 80; this.vx = 0; this.vy = 0; this.angle = 0; this.avel = 0; // tuning values this.mass = 1.0; this.motorForce = 0.18; // acceleration per frame input this.maxSpeed = 8.0; this.waterLinearDamp = 0.05; this.airLinearDamp = 0.01; this.angularDampSubmerged = 0.08; this.angularDampAir = 0.01; this.buoyancyStrength = 0.6; // upward force per submerged sample // buoyancy sample points in local coords (x,y) this.buoyPoints = [ {x:-24,y:10}, {x:24,y:10}, {x:0,y:14} ]; this.width = 64; this.height = 28; this.distance = 0; this.fuel = 100; } update(dt){ // apply motor const throttle = (Input.up ? 1 : 0) - (Input.down ? 1 : 0); // if no fuel -> throttle=0 const actualThrottle = (this.fuel > 0) ? throttle : 0; // engine applies along local x (right) const ax = Math.cos(this.angle) * actualThrottle * this.motorForce; const ay = Math.sin(this.angle) * actualThrottle * this.motorForce; this.vx += ax; this.vy += ay; // tilt control / torque if(Input.left) this.avel -= 0.0009 * 60; if(Input.right) this.avel += 0.0009 * 60; // simple gravity this.vy += 0.008 * 60 * dt; // Buoyancy sampling: count submerged sample points relative to wave height let submergedCount = 0; let submergedDepth = 0; for(const p of this.buoyPoints){ const worldX = this.x + (Math.cos(this.angle)*p.x - Math.sin(this.angle)*p.y); const worldY = this.y + (Math.sin(this.angle)*p.x + Math.cos(this.angle)*p.y); const waterY = Wave.heightAt(worldX); if(worldY > waterY){ // in canvas Y increases downwards; being *below* water means larger Y submergedCount++; submergedDepth += clamp((worldY - waterY)/30, 0, 1); } } const frac = submergedCount / this.buoyPoints.length; if(frac > 0){ // apply upward buoyant force const buoyForce = this.buoyancyStrength * (submergedDepth + frac); this.vy -= buoyForce * dt * 60; // water drag this.vx *= (1 - this.waterLinearDamp); this.vy *= (1 - this.waterLinearDamp); // angular damp this.avel *= (1 - this.angularDampSubmerged); } else { // air drag (less) this.vx *= (1 - this.airLinearDamp); this.vy *= (1 - this.airLinearDamp); this.avel *= (1 - this.angularDampAir); } // clamp speed const speed = Math.hypot(this.vx, this.vy); if(speed > this.maxSpeed){ const s = this.maxSpeed / speed; this.vx *= s; this.vy *= s; } // integrate movement this.x += this.vx; this.y += this.vy; this.angle += this.avel * dt * 60; // simple ground collision (terrain) const groundY = H - Terrain.heightAt(this.x); const hullY = this.y + Math.abs(Math.sin(this.angle)) * 10; if(hullY < groundY - 6){ // above ground -> do nothing } else { // collision response: push up and damp velocities this.y = groundY - 6 - Math.abs(Math.sin(this.angle)) * 10; this.vy = Math.min(this.vy, 0); this.vx *= 0.6; this.avel *= 0.4; } // update distance & fuel this.distance = Math.max(this.distance, Math.floor(this.x / 10)); if(actualThrottle !== 0) this.fuel = Math.max(0, this.fuel - 0.02); // small stabilization: if stuck upside down for long, nudge angle if(Math.abs(this.angle) > Math.PI * 0.66 && frac > 0.5){ // flip gently upright this.avel *= 0.95; this.angle *= 0.98; } } draw(ctx, camX, camY){ ctx.save(); ctx.translate(this.x - camX, this.y - camY); ctx.rotate(this.angle); // hull ctx.fillStyle = '#ff6f3c'; ctx.beginPath(); ctx.moveTo(-32,8); ctx.quadraticCurveTo(0,-18,32,8); ctx.lineTo(32,14); ctx.lineTo(-32,14); ctx.closePath(); ctx.fill(); // cockpit ctx.fillStyle = '#1b3a4b'; ctx.fillRect(-10,-6,20,12); // exhaust flame when throttle if(Input.up && this.fuel>0){ ctx.fillStyle = '#ffd27f'; ctx.beginPath(); ctx.ellipse(-36,10,6,3,0,0,Math.PI*2); ctx.fill(); } ctx.restore(); } } // --------- Game manager ---------- const Game = { boat: new Boat(), last: performance.now(), running: true, toggleWaves(){ Wave.toggle(); document.getElementById('tips').innerText = 'Waves: ' + (Wave.enabled? 'ON' : 'OFF'); }, step(now){ const dt = Math.min(0.033, (now - Game.last) / 1000); Game.last = now; // update Wave.update(dt); Game.boat.update(dt); Camera.follow(Game.boat.x, Game.boat.y); // render ctx.clearRect(0,0,W,H); // draw far sky // water Wave.draw(ctx, Camera.x, Camera.y); // terrain Terrain.draw(ctx, Camera.x, Camera.y); // boat Game.boat.draw(ctx, Camera.x, Camera.y); // UI update document.getElementById('distance').innerText = 'Distance: ' + Game.boat.distance + ' m'; const sp = Math.round(Math.hypot(Game.boat.vx, Game.boat.vy) * 10) / 10; document.getElementById('speed').innerText = 'Speed: ' + sp + ' m/s'; if(Game.running) requestAnimationFrame(Game.step); }, start(){ Game.last = performance.now(); Game.running = true; requestAnimationFrame(Game.step); } }; // start Game.start(); })();
